# Функция вычисления e
function eyler(n)
    s = 0, f = 1
    for k = 1:n
        f *= k
        s += 1/f
    end
    return s
end

# функция вычисления sin(x) с заданной погрешностью
function sin_(x, e)
    s = 0
    a = x
    k = 2
    while abs(a) > e
        s += a
        a *= -x*x / (k*(k+1))
        k += 2
    end
end
